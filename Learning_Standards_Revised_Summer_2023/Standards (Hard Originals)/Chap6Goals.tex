\subsection*{Learning Goals}\hypertarget{1}{}%hypertarget for learning standard referencing
    \begin{enumerate}
       \categorize{VG}{I can visualize the absolute value of the determinant of a $2 \times 2$ and a $3 \times 3$ matrix as a factor by which the corresponding linear transformation changes the area of the unit square or the volume of the unit cube, respectively.}
        \categorize{VG}{I can interpret the sign of the determinant of a matrix geometrically.}
        \categorize{COM}{I can compute the determinant of an arbitrary square matrix.}
        \categorize{COM}{Given a square matrix $A$, I can compute the determinant of $A^{-1}$ without computing $A^{-1}$.}
        \categorize{COM}{Given square matrices $A$ and $B$, I can compute the determinant of $AB$ without computing $AB$.}
        \categorize{CON}{I can use the determinant to determine whether a given linear transformation is invertible.}
        \categorize{COM}{Given a $2 \times 2$ matrix $A$, I can compute the inverse of $A$ using the matrix inverse formula.}
        \categorize{CON}{I can use row reduction to compute the determinant of a matrix}
\end{enumerate}
